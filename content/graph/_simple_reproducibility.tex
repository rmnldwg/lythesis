    \begin{itemize}
        \item \textbf{Graph 1} (no connection btw. II and III): \lytag{simple-nocon-v1}
        \item \textbf{Graph 2} (from II to III): \lytag{simple-IItoIII-v1}
        \item \textbf{Graph 3} (from III to II): \lytag{simple-IIItoII-v1}
        \item \textbf{Graph 4} (both directions): \lytag{simple-both-v1}
    \end{itemize}
